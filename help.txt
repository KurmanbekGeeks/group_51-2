sudo apt update       # Обновить список пакетов
sudo apt upgrade      # Обновить установленные программы

sudo apt install python3  # Установить Python

sudo apt install python3-venv  # Установить venv, если нет

python3 -m venv venv  # Создать окружение

source venv/bin/activate  # Активировать окружение

deactivate  # Выйти из окружения
